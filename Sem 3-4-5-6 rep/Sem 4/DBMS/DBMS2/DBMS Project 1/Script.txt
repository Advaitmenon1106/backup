CREATE TABLE DEPARTMENT(
DName VARCHAR (200),
DNumber INT NOT NULL PRIMARY KEY,
Mgr_ssn INT,
Mgr_start_date VARCHAR (150)
);

CREATE TABLE EMPLOYEE(
Ssn INT NOT NULL PRIMARY KEY,
Fname CHAR(100),
Minit CHAR(100),
Lname CHAR(100),
BDate VARCHAR(100),
Address VARCHAR(250),
Sex VARCHAR (2),
Super_Ssn INT,
Dno INT,
Salary INT,

CONSTRAINT FK_Dno FOREIGN KEY (Dno) REFERENCES DEPARTMENT(DNumber)
);

CREATE TABLE DPT_LOCATIONS(
Dnumber INT,
Dlocation INT,
PRIMARY KEY (Dnumber, Dlocation),
CONSTRAINT FK_DnoLoc FOREIGN KEY(Dnumber) REFERENCES DEPARTMENT(Dnumber)
);

CREATE TABLE PROJECTS(
Pname VARCHAR (200),
Pnumber INT NOT NULL PRIMARY KEY,
PLocation VARCHAR (250),
Dnum INT,
CONSTRAINT FK_Dum FOREIGN KEY(Dnum) REFERENCES DEPARTMENT(Dnumber)
);

CREATE TABLE WORKS_ON(
Ssn INT NOT NULL PRIMARY KEY,
Pno INT,
Hours INT,

CONSTRAINT FK_Pno FOREIGN KEY (Pno) REFERENCES PROJECTS (Pnumber)
);

CREATE TABLE DEPENDENTS(
Ssn INT,
DEPENDENT_NAME VARCHAR(250),
GENDER VARCHAR (250),
Bdate VARCHAR (200),
RELATIONSHIP VARCHAR (250),

CONSTRAINT FK_Essn FOREIGN KEY (Ssn) REFERENCES EMPLOYEE (Ssn)
);
